import java.util.Scanner;
class Book
{
    String title;
    public static void main(String[] args)
    {
        Scanner sc = new Scanner(System.in);
        Book B = new Book();
        System.out.print("enter title");
        B.title = sc.nextLine();
        System.out.println("Book details");
        System.out.println("title : " + B.title);
        sc.close();
    }
}
